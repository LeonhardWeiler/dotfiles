# services.txt - systemd units (re)activated by `./install` after linking
# (self-healing), and deactivated by `--remove-systemd`.
#
# Format: one unit per line, two columns `<scope> <unit>`.
#   scope = user     -> `systemctl --user enable <unit>`
#   scope = system   -> `sudo systemctl enable <unit>`
#   scope = disable  -> `sudo systemctl disable <unit>`  (preset-on unit we want OFF)
# `#` comments and blank lines are ignored.
#
# Only system units that really exist are listed. pipewire/wireplumber are user
# services (package presets, not tracked) and swtpm is socket-activated - those
# have no enable-able bare <name>.service. iptables loads the (default) rules
# from /etc/iptables/; fstrim.timer does periodic SSD TRIM; timesyncd = NTP;
# power-profiles-daemon = power profiles (laptop).
#
# There are currently no `user` units: the battery check and the config sync run
# as plain commands from the dwl autostart (autostart[] in config/dwl/config.h)
# instead of systemd user units.

system  NetworkManager.service
system  getty@tty1.service
system  dnsmasq.service
system  sshd.service
system  legion-conservation.service
system  iptables.service
system  power-profiles-daemon.service
system  systemd-timesyncd.service
system  fstrim.timer

# libvirt: run the modular, socket-activated daemons instead of the monolithic
# libvirtd.service - the sockets start virtqemud/virtnetworkd/virtstoraged on the
# first `qemu:///system` connection, so nothing blocks boot. (These are usually
# preset-enabled already; listed to make the intent explicit and reproducible.)
system  virtqemud.socket
system  virtnetworkd.socket
system  virtstoraged.socket

# Boot-time trims (disabled to cut startup latency):
#   libvirtd.service              - monolithic libvirt daemon, ~1.8s on the
#                                   critical path; replaced by the sockets above.
#   NetworkManager-wait-online    - blocks network-online.target until a link is
#                                   up; pointless on a laptop with NM.
disable  libvirtd.service
disable  NetworkManager-wait-online.service
